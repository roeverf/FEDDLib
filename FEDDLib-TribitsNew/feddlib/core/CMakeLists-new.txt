#
# A) Define the package
#
tribits_package( core  ENABLE_SHADOWING_WARNINGS  CLEANED )
SET(${PACKAGE_NAME}_VERSION "1.0.0")

#
# B) Platform-specific checks
#
include(CheckFor__int64)
check_for___int64(HAVE_CORE___INT64)
tribits_pkg_export_cache_var(HAVE_CORE___INT64)

#
# C) Set up package-specific options
#
tribits_add_debug_option()
tribits_add_show_deprecated_warnings_option()

FOREACH(TRILINOS_PACKAGE_NAME in ${Trilinos_PACKAGE_LIST})
    IF(${TRILINOS_PACKAGE_NAME} STREQUAL "NOX")
        SET(FEDD_HAVE_NOX TRUE)
    ENDIF()
ENDFOREACH()

FOREACH(TRILINOS_PACKAGE_NAME in ${Trilinos_PACKAGE_LIST})
    IF(${TRILINOS_PACKAGE_NAME} STREQUAL "Teko")
        SET(FEDD_HAVE_TEKO TRUE)
    ENDIF()
ENDFOREACH()
TRIBITS_CONFIGURE_FILE(${PACKAGE_NAME}_config.h)
#
# D) Add the libraries, tests, and examples
#
add_subdirectory(LinearAlgebra)
tribits_add_test_directories(LinearAlgebra/test)

#
# E) Do standard post processing
#
tribits_package_postprocess()
